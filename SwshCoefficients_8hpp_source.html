23 // The Coefficients class acts largely as a memory-safe container for a
24 // `sharp_alm_info*`, required for use of libsharp transform utilities.
25 // The libsharp utilities are currently constructed to only provide user
26 // functions with collocation data for spin-weighted functions and
27 // derivatives. If and when the libsharp utilities are expanded to provide
28 // spin-weighted coefficients as output, this class should be expanded to
29 // provide information about the value and storage ordering of those
30 // coefficients to user code. This should be implemented as an iterator, as is
31 // done in SwshCollocation.hpp.
32 class Coefficients {
33  public:
34  explicit Coefficients(size_t l_max) noexcept;
35 
36  ~Coefficients() = default;
37  Coefficients() = default;
38  Coefficients(const Coefficients&) = delete;
39  Coefficients(Coefficients&&) = default;
40  Coefficients& operator=(const Coefficients&) = delete;
41  Coefficients& operator=(Coefficients&&) = default;
42  sharp_alm_info* get_sharp_alm_info() const noexcept {
43  return alm_info_.get();
44  }
45 
46  size_t l_max() const noexcept { return l_max_; }
47 
48  private:
49  std::unique_ptr<sharp_alm_info, DestroySharpAlm> alm_info_;
50  size_t l_max_ = 0;
51 };
